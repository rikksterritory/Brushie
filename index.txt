  <script>
    // Simple 2-minute countdown with three states: idle â†’ brushing â†’ finished
    const startBtn = document.getElementById('start');
    const timeEl = document.getElementById('time');
    const statusEl = document.getElementById('status');

    let remaining = 120;   // seconds
    let tick = null;       // interval id
    let state = 'idle';    // 'idle' | 'brushing' | 'finished'

    function render() {
      const m = String(Math.floor(remaining / 60)).padStart(2, '0');
      const s = String(remaining % 60).padStart(2, '0');
      timeEl.textContent = `${m}:${s}`;

      if (state === 'idle') {
        startBtn.textContent = 'Start Brushing';
        statusEl.textContent = 'Idle';
      } else if (state === 'brushing') {
        startBtn.textContent = 'Pause';
        statusEl.textContent = 'Brushingâ€¦';
      } else {
        startBtn.textContent = 'Restart';
        statusEl.textContent = 'Done! Minty fresh âœ¨';
      }
    }

    function start() {
      if (state === 'finished') remaining = 120;
      state = 'brushing';
      render();
      tick = setInterval(() => {
        remaining--;
        if (remaining <= 0) {
          clearInterval(tick);
          tick = null;
          remaining = 0;
          state = 'finished';
          render();
          return;
        }
        render();
      }, 1000);
    }

    function pause() {
      clearInterval(tick);
      tick = null;
      state = 'idle';
      render();
    }

    startBtn.addEventListener('click', () => {
      if (state === 'idle') start();
      else if (state === 'brushing') pause();
      else if (state === 'finished') start(); // restart
    });

    render();
  </script>
